def evalArrhenius(autoTSTExpression):
    try:
        my_string = str(autoTSTExpression)[1:]
        result = eval(my_string, {}, {'Arrhenius': rmgpy.kinetics.Arrhenius})
    except:
        result = autoTSTExpression
    return result
